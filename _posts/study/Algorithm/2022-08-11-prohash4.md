```py
#위장
def solution(clothes):
    wear={}
    for i in clothes:
        if i[1] not in wear:
            wear[i[1]]=[i[0]]
            print(wear)
        else:
            wear[i[1]].append(i[0])
    
    answer = 1
    for i in wear.keys():
        answer*=(len(wear[i])+1)
        
    return answer-1
```
